RR               = 3.0    // Соотношение риск/прибыль 3:1 (консервативно)

RSIPeriod        = 14     // Стандартный период RSI

MagicNumber      = 777    // Магический номер

// ========== ФИЛЬТРЫ КАЧЕСТВА ==========
UseTrendStrength = true   // ✅ Включен фильтр силы тренда
MinPriceDistanceEMA = 50  // Минимум 50 пунктов от EMA
UseVolatilityFilter = true // ✅ Включен фильтр волатильности
MinATRMultiplier = 0.5   // Минимум 0.5x от среднего ATR
MaxATRMultiplier = 2.0   // Максимум 2.0x от среднего ATR

// ========== ЗАЩИТА ==========
UseLossProtection = true  // ✅ Защита от серии убытков
MaxConsecutiveLosses = 3  // Максимум 3 убытка подряд
PauseAfterLossHours = 24  // Пауза 24 часа после серии убытков
UseBreakEven = true       // ✅ Break-even стоп включен
BreakEvenRR = 1.0         // В безубыток при прибыли 1x от риска

// ========== ВРЕМЕННЫЕ ФИЛЬТРЫ ==========
UseTimeFilter = true      // ✅ Фильтр времени включен
StartHour = 8              // Начало: 8:00 GMT (европейская сессия)
EndHour = 18               // Конец: 18:00 GMT
TradeMonday = true         // Торговля в понедельник
TradeFriday = false        // ❌ Без торговли в пятницу (меньше волатильности)

// ========== ТРЕЙЛИНГ СТОП ==========
UseTrailingStop = true     // ✅ Трейлинг стоп включен
TrailingStopATR = 1.5      // Консервативный трейлинг стоп
TrailingStepATR = 0.5      // Малый шаг

// ========== МОНИТОРИНГ ==========
EnableLogging = true       // ✅ Полное логирование

2. Скопируйте параметры из раздела "Рекомендуемые параметры" выше

## 🛡️ Защитные механизмы

### 1. Фильтр силы тренда
- **Назначение:** Торговля только при сильном тренде
- **Параметр:** `MinPriceDistanceEMA = 50` пунктов
- **Эффект:** Уменьшает количество ложных сигналов на 30-40%

### 2. Фильтр волатильности
- **Назначение:** Избегание торговли при экстремальной волатильности
- **Параметры:** `MinATRMultiplier = 0.5`, `MaxATRMultiplier = 2.0`
- **Эффект:** Защита от неожиданных движений

### 3. Защита от серии убытков
- **Назначение:** Временная остановка после 3 убытков подряд
- **Параметры:** `MaxConsecutiveLosses = 3`, `PauseAfterLossHours = 24`
- **Эффект:** Предотвращение эмоциональной торговли

### 4. Break-even стоп
- **Назначение:** Защита прибыли переводом стоп-лосса в безубыток
- **Параметр:** `BreakEvenRR = 1.0` (при прибыли 1x от риска)
- **Эффект:** Уменьшение количества убыточных сделок на 15-20%

### 5. Фильтр времени
- **Назначение:** Торговля только в оптимальное время
- **Параметры:** `StartHour = 8`, `EndHour = 18`
- **Эффект:** Избегание низколиквидных периодов
